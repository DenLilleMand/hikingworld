CREATE SCHEMA hikingworld;
USE hikingworld;
CREATE TABLE account (id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY, username VARCHAR(145) NOT NULL UNIQUE, password VARCHAR(145) NOT NULL, salt VARCHAR(145), verification TINYINT(1), checksum VARCHAR(145), created DATETIME);
CREATE TABLE post (id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY, text VARCHAR(5000) NOT NULL);
ALTER TABLE post CHANGE text description VARCHAR(1024)
CREATE TABLE attempts (id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY, username VARCHAR(145) NOT NULL UNIQUE, attempts INT(11), lastlogin DATETIME);
